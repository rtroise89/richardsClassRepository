    null_dist <- c()

Correlation

    corr_dist <- c()
    
    for(i in 1:10000) {
      x <- rnorm(10,0,1)
      y <- rnorm(10,0,1)
      corr_dist <- c(corr_dist, cor(x,y))
    }

    plot(density(corr_dist))

F-distribution

    Subjects <- rep(1:4,3)
    df2 <- 8
    df1 <- 2

    F_dist <- c()
    
    for(i in 1:10000) {
      A <- rnorm(4,0,1)
      B <- rnorm(4,0,1)
      C <- rnorm(4,0,1)
      Mean_subject <- c(A,B,C)
      
      corr2_dist <- cor(Subjects,Mean_subject)
      F_dist <- c(F_dist, (corr2_dist/(1-corr2_dist))*(df2/df1))
    }

    plot(density(F_dist))

critical value

    z <- density(F_dist)

    f_1 <- which(z$y >= 0.05)
    z$x[f_1[length(f_1)]]
## [1] 2.936103
